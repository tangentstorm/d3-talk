#+begin_src js

// allocate empty <td> cells with appropriate classes
rows.each(function(dx){
  let cells = d3.select(this).selectAll('td').data(xs);
  cells.enter().append('td');
  cells.attr('class',
    dy => vs[dx][dy] == undefined ? 'nil' : 'val');
});
